## 2. Surgical Constraints (attach to EVERY subagent prompt)

1. **Backend is off-limits.** Do not touch `server/` or any `shared/` schema file. If a change seems to require a backend edit, stop and return the file path so the orchestrator can reassign.
2. **No UI redesign.** Only data wiring, conditionals, and helper swaps. Preserve existing CSS classes, element structure, and copy. Button labels only change where explicitly listed.
3. **Name fallback chain** (apply everywhere a "primary label" is shown):
   ```
   identifiedName
     → `${identifiedBrand ?? ""} ${identifiedModel ?? ""}`.trim() when both present
     → category (only when non-weak, i.e. not "Uncategorized")
     → "Needs review"
   ```
   Never fall through to `itemName` when `itemName === "Scanned Item"` or empty.
4. **Never use "Scanned Item" as a primary label.** Existing uses in
   `formatItemDisplay.ts` treat it as a *weak* sentinel, which is fine.
   Occurrences that *write* `"Scanned Item"` (e.g. `RoomDetailView.tsx:1661`)
   must be replaced with `undefined`/omitted — backend accepts empty itemName
   for photo-first intake.
5. **Derived intent + lifecycle.** Because the API does not yet emit `intent`
   or `lifecycleStatus` (see §0), compute them client-side in
   `client/src/utils/itemStatus.ts` and expose them via pure helpers. UI code
   must call those helpers rather than reading raw `status` / `keepFlag` /
   `willingToSell`. This is the **single seam** a future backend phase will
   swap.

### 2.1 Derivation rules for `intent` (client-side)

```
intent(item) =
  item.status === "SOLD"                              → "sold"     (completed)
  item.status === "DONATED"                           → "donated"  (completed)
  item.status === "SHIPPED"                           → "shipped"  (completed)
  item.status === "DISCARDED"                         → "discarded"(completed)
  item.status === "LISTED"                            → "sell"
  item.status === "KEPT"         OR item.keepFlag     → "keep"
  item.recommendation === "SHIP" AND item.status === "REVIEWED" → "ship"
  item.recommendation === "DONATE" AND item.status === "REVIEWED" → "donate"
  item.willingToSell === true                         → "sell"     (soft)
  otherwise                                           → "undecided"
```

### 2.2 Derivation rules for `lifecycleStatus` (client-side)

```
lifecycleStatus(item) =
  item.status ∈ {SOLD, DONATED, SHIPPED, DISCARDED}   → "completed"
  item.status ∈ {LISTED, KEPT, REVIEWED, STORED}      → "planned"
  otherwise (UNREVIEWED)                              → "undecided"
```

`completedAt` is already emitted by the server (`types.ts:247`) and can be used
directly as a stronger signal where present.

#### A.3 `client/src/utils/itemStatus.ts`  — NEW HELPERS

```ts
import type { Item, ItemIntent, LifecycleStatus, ItemStatus } from "../types";

const COMPLETED_STATUSES: ReadonlySet<ItemStatus> = new Set([
  "SOLD", "DONATED", "SHIPPED", "DISCARDED",
]);
const PLANNED_STATUSES: ReadonlySet<ItemStatus> = new Set([
  "LISTED", "KEPT", "REVIEWED", "STORED",
]);

export function itemLifecycle(item: Pick<Item, "status">): LifecycleStatus {
  if (COMPLETED_STATUSES.has(item.status)) return "completed";
  if (PLANNED_STATUSES.has(item.status))   return "planned";
  return "undecided";
}

export function itemIntent(
  item: Pick<Item, "status" | "recommendation" | "keepFlag" | "willingToSell">,
): ItemIntent {
  switch (item.status) {
    case "SOLD":      return "sold";
    case "DONATED":   return "donated";
    case "SHIPPED":   return "shipped";
    case "DISCARDED": return "discarded";
    case "LISTED":    return "sell";
    case "KEPT":      return "keep";
  }
  if (item.keepFlag) return "keep";
  if (item.status === "REVIEWED") {
    if (item.recommendation === "SHIP")   return "ship";
    if (item.recommendation === "DONATE") return "donate";
    if (item.recommendation === "SELL_NOW" || item.recommendation === "SELL_SOON") return "sell";
    if (item.recommendation === "KEEP") return "keep";
  }
  if (item.willingToSell) return "sell";
  return "undecided";
}

export function isCompletedItem(item: Pick<Item, "status">): boolean {
  return itemLifecycle(item) === "completed";
}

export function isPlannedItem(item: Pick<Item, "status">): boolean {
  return itemLifecycle(item) === "planned";
}

export function isUndecidedItem(item: Pick<Item, "status">): boolean {
  return itemLifecycle(item) === "undecided";
}

/**
 * Name fallback chain per Phase 3 contract:
 *   identifiedName → "brand model" → category (non-weak) → "Needs review".
 * NEVER falls through to itemName when itemName === "Scanned Item" or empty.
 */
export function itemPrimaryLabel(
  item: Pick<Item, "identifiedName" | "identifiedBrand" | "identifiedModel" | "category" | "itemName">,
): { label: string; weak: boolean } {
  const name = item.identifiedName?.trim();
  if (name) return { label: name, weak: false };

  const brand = item.identifiedBrand?.trim();
  const model = item.identifiedModel?.trim();
  const bm = [brand, model].filter(Boolean).join(" ").trim();
  if (bm) return { label: bm, weak: false };

  const legacyName = item.itemName?.trim();
  if (legacyName && legacyName !== "Scanned Item") {
    return { label: legacyName, weak: false };
  }

  const cat = item.category?.trim();
  if (cat && cat !== "Uncategorized") return { label: cat, weak: true };

  return { label: "Needs review", weak: true };
}
```
